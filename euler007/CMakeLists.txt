cmake_minimum_required(VERSION 3.1.0 FATAL_ERROR)
project(Euler7)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wpedantic -Wextra -Weffc++ -g")

include_directories(include)
file(GLOB EULER7_SF "src/*")

add_library(euler7 SHARED ${EULER7_SF})

enable_testing()
find_package(GTest REQUIRED)

include_directories(${GTEST_INCLUDE_DIRS})

add_executable(Euler7Spec test/Euler7Spec.cpp)
target_link_libraries(Euler7Spec euler7 ${GTEST_BOTH_LIBRARIES})
